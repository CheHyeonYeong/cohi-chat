## 참고 - Python 백엔드 원본
```python
@router.get("/time-slots/{host_username}")
async def get_host_timeslots(host_username: str, session: DbSessionDep) -> list[TimeSlotOut]:
    # 호스트 조회
    host = session.scalar_one_or_none()
    if host is None or host.calendar is None:
        raise HostNotFoundError()

    # 타임슬롯 목록 조회
    return session.scalars(select(TimeSlot).where(TimeSlot.calendar_id == host.calendar.id)).all()
```
